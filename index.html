<body>

    <h1>과일 선택 게임</h1>

    <p>이름을 입력하고, 원하는 과일을 선택하세요! (각 과일은 최대 4명까지만 선택 가능)</p>

    <input type="text" id="username" placeholder="이름을 입력하세요" required>


    <div>

        <span class="fruit" onclick="selectFruit('딸기', '🍓')">🍓</span>

        <span class="fruit" onclick="selectFruit('바나나', '🍌')">🍌</span>

        <span class="fruit" onclick="selectFruit('사과', '🍎')">🍎</span>

        <span class="fruit" onclick="selectFruit('포도', '🍇')">🍇</span>

        <span class="fruit" onclick="selectFruit('블루베리', '🫐')">🫐</span>

    </div>


    <div class="result" id="resultArea" style="display: none;">

        <!-- 결과가 여기에 표시됩니다 -->

    </div>


    <script>

        const maxSelection = 4;

        const fruits = {

            '딸기': ['피넛'],

            '바나나': ['리아'],

            '사과': ['애니스'],

            '포도': ['레인'],

            '블루베리': ['프레디']

        };


        const selectedUsers = ['피넛', '리아', '애니스', '레인', '프레디'];


        function selectFruit(fruit, emoji) {

            const username = document.getElementById("username").value.trim();

            if (username === "") {

                alert("이름을 입력해주세요!");

                return;

            }


            if (selectedUsers.includes(username)) {

                alert(`${username}님은 이미 과일을 선택하셨습니다.`);

                return;

            }


            if (fruits[fruit].length < maxSelection) {

                fruits[fruit].push(username);

                selectedUsers.push(username);

                alert(`${username}님이 ${fruit}(${emoji})를 선택했습니다!`);

                showResults();

            } else {

                alert(`${fruit}(${emoji})는 이미 ${maxSelection}명이 선택했습니다!`);

            }


            document.getElementById("username").value = "";

        }


        function showResults() {

            let resultText = "<h2>과일 선택 결과</h2>";

            for (const fruit in fruits) {

                resultText += `<p>${fruit}: ${fruits[fruit].join(", ")}</p>`;

            }

            const resultArea = document.getElementById("resultArea");

            resultArea.innerHTML = resultText;

            resultArea.style.display = "block";

        }

    </script>
